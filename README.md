# bootstrap-codeigniter
Helper para manejar bootstrap 3 en las vistas de codeigniter 

<h1> Documentation </h1>
	Inside helper

<h1> Example </h1>

<h3> Controller </h3>

	public function index()
	{	
		//load helper bootstrap
		$this->load->helper('bootstrap');
		//call view
		$this->load->view('yourview');		
	}
